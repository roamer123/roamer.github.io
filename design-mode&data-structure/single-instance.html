单例模式

单例模式

  • 定义： 保证一个类仅有一个实例，并提供一个访问它的全局访问。

  • 应用：线程池，全局缓存，window对象等，在前端点击登录按钮弹出悬浮框，无论点击多少次悬浮框都是唯一的，而写只会被创建一次。类似的还有alert,message等组件，全局使用时，dom对象只创建一次，事件绑定，函数执行多次但是绑定只有一次

  • 实现方式：代理实现单例，每次创建时通过代理处理决定返回已有实例还是创建实例对象

应用：单个dom实例

  • 透明单例模式
var CreateDiv = (function() {
    var instance;

    var CreateDiv = function(html) {
        if(instance) {
            return instance;
        }
        this.html = html;
        this.init();

        return instance = this;
    }

    CreateDiv.prototype.init = function() {
        var div = document.creatElement('div');
        div.innerHTML = this.html;
        document.body.appendChild(div)
    }

    return CreateDiv
})();

闭包，保存自定义的实例

var CreateDiv = (function() {
    var CreateDiv = function(html) {
        this.html = html;
        this.init();

        return instance = this;
    }

    CreateDiv.prototype.init = function() {
        var div = document.creatElement('div');
        div.innerHTML = this.html;
        document.body.appendChild(div)
    }
})

var ProxySingletonCreateDiv = (function() {
    var instance;
    return function(html) {
        if(!instance) {
            instance = new CreateDiv(html);
        }

        return instance;
    }
})

// namespace
var MyApp = {};

MyApp.namespace = function(name) {
    var parts = name.split('.');
    var current = MyApp;
    for(var i in parts) {
        if(!current[parts[i]]) {
            current[parts[i]] = {}
        }
        current = current[parts[i]];
    }
};

MyApp.namespace('a.aa')
MyApp.namespace('dom')
