模板方法模式

模板方法模式

模板方法主要有两个结构组成：1.抽象父类， 2，具体实现的子类。（java抽象类）

钩子方法

const Beverage = function(){}

Beverage.prototype.boilWater = function(){
    console.log('把水煮沸');
};

Beverage.prototype.brew = function(){
    throw new Error('子类必须重写brew方法')
};

Beverage.prototype.pourInCup = function(){
    throw new Error('子类必须重写pourInCup方法')
};

Beverage.prototype.addCondiments = function(){
    throw new Error('子类必须重写addCondiments方法')
};

Beverage.prototype.customerWantsCondiments = function(){
    return true;
};

Beverage.prototype.init = function(){
    this.boilWater();
    this.brew();
    this.pourInCup();
    if(this.customerWantsCondiments()){
        this.addCondiments();
    };
};


const CoffeeWithHook = function(){};

CoffeeWithHook.prototype = new Beverage();

CoffeeWithHook.prototype.brew = function(){
    console.log('用沸水冲泡咖啡');
}

CoffeeWithHook.prototype.pourInCup = function(){
    console.log('把咖啡倒进杯子');
}

CoffeeWithHook.prototype.addCondiments = function(){
    console.log('加糖和牛奶');
}

CoffeeWithHook.prototype.brew = function(){
    return window.confirm('请问需要调料？')
}

const coffeeWithHook = new CoffeeWithHook();

coffeeWithHook.init();

好莱坞原则——应用发布-订阅模式，回调函数。

const Beverage = function(param){

    const boilWater = function(){
        console.log('把水煮沸');
    };

    const brew = function(){
        throw new Error('必须传递brew方法')
    };

    const pourInCup = function(){
        throw new Error('必须传递pourInCup方法')
    };

    const addCondiments = function(){
        throw new Error('必须传递addCondiments方法')
    };

    const F = function(){};

    F.prototype.init = function(){
        boilWater();
        brew();
        pourInCup();
        addCondiments();
    }

    return F;
}

const Coffee = new Beverage({
    brew: function(){
        console.log('用水冲泡咖啡');
    },
    pourInCup: function(){
        console.log('把咖啡倒进杯子');
    },
    addCondiments: function(){
        console.log('加糖和牛奶');
    }
});

const Tea = new Beverage({
    brew: function(){
        console.log('用水浸泡茶叶');
    },
    pourInCup: function(){
        console.log('把茶倒进杯子');
    },
    addCondiments: function(){
        console.log('加柠檬');
    }
});

const coffee = new Coffee();
coffee.init();

const tea = new Tea();
tea.init();
